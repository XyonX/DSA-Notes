bool checkAnagrams(string a, string b){
    
    //base case to check size mismatch
    if(a.size() !=b.size())return false;
    
    vector<int>fA(26,0);
    vector<int>fB(26,0);

    for( int i=0;i<a.size();i++){
        fA[a[i]-'a']++;
        fB[b[i]-'a']++;
    }
    
    for(char c : a){
        if(fA[c-'a'] != fB[c-'a'])return false;
    }
    return true;
    
    
    
    
}

bool isPalindrome(string s){
    int n=s.size();
    for(int i=0;i<n/2;i++){
        if(tolower(s[i])!= tolower(s[n-i-1]))return false;
    }
    return true;
}

bool isP(int i, string& s){
    
    if(i>s.size()/2){
        return true;
    }
    if(tolower(s[i])!= tolower(s[s.size()-i-1]))return false;
    return isP(i+1,s);
}
